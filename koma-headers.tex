% Configuration for running headers and footers using KOMA-Script.
% Requires: KOMA-Script document class (scrreprt, scrbook, or scrartcl)

% Verify KOMA-Script document class is loaded
\makeatletter
\@ifundefined{KOMAClassName}{%
  \PackageError{koma-headers}{KOMA-Script document class required}{%
    This file requires a KOMA-Script document class (scrreprt, scrbook, or scrartcl).%
  }%
}{}
\makeatother

% KOMA-Script headers using scrlayer-scrpage (replaces fancyhdr)
\usepackage{scrlayer-scrpage}

% Headers: Chapter/section names centered
\automark[section]{chapter}  % [right-mark]{left-mark}
\cehead{\headmark}           % Center Even HEAD: chapter name
\cohead{\headmark}           % Center Odd HEAD: section name
